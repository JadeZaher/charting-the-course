# Edge Function Secrets Migration File
# 
# This file can be used to migrate secrets from source to target project.
# 
# Usage:
#   1. Copy this file to secrets.txt
#   2. Fill in your secret values (one per line, format: NAME=value)
#   3. Run: ./scripts/migrate-supabase-cli.sh
#
# Or set environment variables:
#   export SECRET_SECRET_NAME=value
#
# Format: SECRET_NAME=value
# 
# Common secrets for Supabase Edge Functions:
# SUPABASE_URL=https://your-project.supabase.co
# SUPABASE_ANON_KEY=your-anon-key
# SUPABASE_SERVICE_ROLE_KEY=your-service-role-key
#
# Add your custom secrets below:
# OPENAI_API_KEY=your-openai-key
# STRIPE_SECRET_KEY=your-stripe-key
# CUSTOM_API_KEY=your-custom-key
